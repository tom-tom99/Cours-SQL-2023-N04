    select ID from FOURNISSEUR where NOM LIKE 'Française d''Imports')

select * from ARTICLE 
JOIN FOURNISSEUR ON ARTICLE.ID_FOU = FOURNISSEUR.ID
where FOURNISSEUR.NOM IN ('Française d''Imports', 'Dubois & Fils');

--h)

    select ID from FOURNISSEUR 
    where NOM NOT IN ('Française d''Imports', 'Dubois & Fils'));
