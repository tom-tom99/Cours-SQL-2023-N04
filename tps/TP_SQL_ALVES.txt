CREATE TABLE compta.ARTICLE (
	ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'PK de ARTICLE' , 
	REF VARCHAR(10) UNIQUE NOT NULL COMMENT 'Réf interne de l\'ARTICLE' , 
	DESIGNATION VARCHAR(50) NOT NULL COMMENT 'nom de l\'article' , 
	PRIX DECIMAL(6,2) NOT NULL COMMENT 'prix de l\'article' , 
	ID_FOU INT(10) NOT NULL COMMENT 'ID du fournisseur' 
);
CREATE TABLE compta.FOURNISSEUR (
	ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'PK de fournisseur',
	NOM VARCHAR(10) NOT NULL COMMENT 'nom fournisseur'
);
CREATE TABLE compta.BON (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'PK du Bon Commande' ,
    NUMERO INT(10) UNIQUE NOT NULL COMMENT 'Réf comptable du BC' ,
    DATE_CMDE DATE NOT NULL COMMENT 'Date de la commande' ,
    DELAI INT(4) COMMENT 'Délai de livraison (en jours)' ,
    ID_FOU INT(10) NOT NULL COMMENT 'ID du fournisseur' 
);
CREATE TABLE compta.COMPO (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT ,
    QTE INT(4) NOT NULL COMMENT 'quantité d\'articles commandés' ,
    ID_ART INT(10) NOT NULL ,
    ID_BON INT(10) NOT NULL
);
