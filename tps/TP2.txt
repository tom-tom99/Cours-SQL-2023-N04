CREATE TABLE ARTICLE(
ID INT PRIMARY KEY,
REF VARCHAR(30),
DESIGNATION VARCHAR(30),
PRIX DECIMAL,
ID_FOU VARCHAR(30)
);

CREATE TABLE FOURNISSEUR (
ID VARCHAR(30) PRIMARY KEY,
NOM VARCHAR(30)
);

CREATE TABLE BON (
ID INT PRIMARY KEY,
NUMERO VARCHAR(30),
DATE_CMDE DATE,
DELAI TIME,
ID_FOU VARCHAR(30)
);


CREATE TABLE COMPO(
	ID INT PRIMARY KEY AUTO_INCREMENT,
	QTE INT,
	ID_ART INT,
    	FOREIGN KEY (ID_ART)
	    REFERENCES ARTICLE(ID),
	ID_BON INT,
    	FOREIGN KEY (ID_BON)
	    REFERENCES BON(ID)
)
